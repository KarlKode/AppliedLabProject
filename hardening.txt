14.
sudo gedit /etc/rsyslog.conf
(client)
# certificate files - just CA for a client
$DefaultNetstreamDriverCAFile ./contrib/gnutls/ca.pem

# set up the action
$DefaultNetstreamDriver gtls # use gtls netstream driver
$ActionSendStreamDriverMode 1 # require TLS for the connection
$ActionSendStreamDriverAuthMode anon # server is NOT authenticated
*.* @@(o)10.10.10.100:10514 # send (all) messages

(server)
# make gtls driver the default
$DefaultNetstreamDriver gtls

# certificate files
$DefaultNetstreamDriverCAFile ./contrib/gnutls/ca.pem
$DefaultNetstreamDriverCertFile ./contrib/gnutls/cert.pem
$DefaultNetstreamDriverKeyFile ./contrib/gnutls/key.pem

$ModLoad imtcp # load TCP listener

$InputTCPServerStreamDriverMode 1 # run driver in TLS-only mode
$InputTCPServerStreamDriverAuthMode anon # client is NOT authenticated
$InputTCPServerRun 10514 # start up listener at port 10514
